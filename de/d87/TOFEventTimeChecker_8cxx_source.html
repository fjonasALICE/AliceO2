97 void TOFEventTimeChecker::fillMatching(GID gid)
98 {
99  if (!gid.includesDet(DetID::TOF)) {
100  return;
101  }
102  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
103  const o2::track::TrackLTIntegral& info = match.getLTIntegralOut();
104 
105  MyTrack trk;
106  int trksource = 5;
107  if (gid.getSource() == GID::TPCTOF) {
108  const auto& array = mRecoData.getTPCTracks();
109  GID gTrackId = match.getTrackRef();
110  const auto& srctrk = array[gTrackId.getIndex()];
111  trk.mPt = srctrk.getPt();
112  trk.mP = srctrk.getP();
113  trksource = 0;
114  } else if (gid.getSource() == GID::ITSTPCTOF) {
115  const auto& array = mRecoData.getTPCITSTracks();
116  GID gTrackId = match.getTrackRef();
117  const auto& srctrk = array[gTrackId.getIndex()];
118  trk.mPt = srctrk.getPt();
119  trk.mP = srctrk.getPt();
120  trksource = 1;
121  } else if (gid.getSource() == GID::TPCTRDTOF) {
122  const auto& array = mRecoData.getTPCTRDTracks<o2::trd::TrackTRD>();
123  GID gTrackId = match.getTrackRef();
124  const auto& srctrk = array[gTrackId.getIndex()];
125  trk.mPt = srctrk.getPt();
126  trk.mP = srctrk.getPt();
127  trksource = 2;
128  } else if (gid.getSource() == GID::ITSTPCTRDTOF) {
129  const auto& array = mRecoData.getITSTPCTRDTracks<o2::trd::TrackTRD>();
130  GID gTrackId = match.getTrackRef();
131  const auto& srctrk = array[gTrackId.getIndex()];
132  trk.mPt = srctrk.getPt();
133  trk.mP = srctrk.getPt();
134  trksource = 3;
135  }
136 
137  const char* sources[5] = {"TPC", "ITS-TPC", "TPC-TRD", "ITS-TPC-TRD", "NONE"};
138 
139  trk.expTimes[0] = info.getTOF(2); // pi
140  trk.expTimes[1] = info.getTOF(3); // ka
141  trk.expTimes[2] = info.getTOF(4); // pr
142  trk.expSigma[0] = 120; // dummy resolution (to be updated)
143  trk.expSigma[1] = 120; // dummy resolution (to be updated)
144  trk.expSigma[2] = 120; // dummy resolution (to be updated)
145 
146  // int tofcl = match.getIdxTOFCl();
147  // trk.mSignal = mTOFClustersArrayInp[tofcl].getTime();
148  trk.mSignal = match.getSignal();
149  trk.mTOFChi2 = match.getChi2();
150  trk.mLength = info.getL();
151  // trk.mHypo = 0;
152 
153  mMyTracks.push_back(trk);
154 }

166 void TOFEventTimeChecker::run(ProcessingContext& pc)
167 {
168  mTimer.Start(false);
169 
170  mMyTracks.clear();
171  mRecoData.collectData(pc, *mDataRequest.get());
172 
173  mIsTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF));
174  mIsITSTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF));
175  mIsITSTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF));
176  mIsTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF));
177 
178  mTOFClustersArrayInp = mRecoData.getTOFClusters();
179 
180  LOG(DEBUG) << "isTrackSourceLoaded: TPC -> " << mIsTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ")";
181  LOG(DEBUG) << "isTrackSourceLoaded: ITSTPC -> " << mIsITSTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ")";
182  LOG(DEBUG) << "isTrackSourceLoaded: TPCTRD -> " << mIsTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ")";
183  LOG(DEBUG) << "isTrackSourceLoaded: ITSTPCTRD -> " << mIsITSTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ")";
184  LOG(DEBUG) << "TOF cluster size = " << mTOFClustersArrayInp.size();
185 
186  if (!mTOFClustersArrayInp.size()) {
187  return;
188  }
189 
190  auto creator = [this](auto& trk, GID gid, float time0, float terr) {
191  this->fillMatching(gid);
192  return true;
193  };
194  mRecoData.createTracksVariadic(creator);
195 
196  // sorting matching in time
197  std::sort(mMyTracks.begin(), mMyTracks.end(),
198  [](MyTrack a, MyTrack b) { return a.tofSignal() < b.tofSignal(); });
199 
200  for (auto& element : mMyTracks) { // loop print
201  LOG(INFO) << "Time cluster = " << element.tofSignal() << " ps - pt = " << element.pt();
202  }
203 
204  std::vector<MyTrack> tracks;
205  for (int i = 0; i < mMyTracks.size(); i++) { // loop looking for interaction candidates
206  tracks.clear();
207  int ntrk = 1;
208  double time = mMyTracks[i].tofSignal();
209  tracks.emplace_back(mMyTracks[i]);
210  for (; i < mMyTracks.size(); i++) {
211  double timeCurrent = mMyTracks[i].tofSignal();
212  if (timeCurrent - time > 25E3) {
213  i--;
214  break;
215  }
216  tracks.emplace_back(mMyTracks[i]);
217  ntrk++;
218  }
219  if (ntrk > 2) { // good candidate with time
220  processEvent(tracks);
221  }
222  }
223 
224  mTimer.Stop();
225 }
